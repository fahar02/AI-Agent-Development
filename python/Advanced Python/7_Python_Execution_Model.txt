1. INTRODUCTION TO PYTHON EXECUTION
================================================================================

Python is an interpreted language, but it's not directly interpreted from source
code. Instead, Python source code is compiled to bytecode, which is then
executed by the Python Virtual Machine (PVM).

Execution Flow:
1. Source code (.py file)
2. Compilation to bytecode (.pyc file)
3. Execution by Python Virtual Machine
4. Output/Results

Key Concepts:
- Bytecode: Intermediate representation between source and machine code
- PVM: Executes bytecode instructions
- Compilation: Converts source to bytecode
- Caching: Bytecode is cached in __pycache__ directory


================================================================================
2. BYTECODE
================================================================================

Bytecode is an intermediate representation of Python code. It's platform-
independent and can be executed by any Python interpreter.

VIEWING BYTECODE:
-----------------

import dis

def add(a, b):
    return a + b

# Disassemble function to see bytecode
dis.dis(add)

# Output:
#   2           0 LOAD_FAST                0 (a)
#               2 LOAD_FAST                1 (b)
#               4 BINARY_ADD
#               6 RETURN_VALUE


BYTECODE INSTRUCTIONS:
----------------------

import dis

# Simple arithmetic
def arithmetic():
    x = 5
    y = 10
    z = x + y
    return z

dis.dis(arithmetic)

# Output shows bytecode instructions:
# LOAD_CONST: Load constant value
# STORE_FAST: Store in local variable
# LOAD_FAST: Load from local variable
# BINARY_ADD: Add two values
# RETURN_VALUE: Return from function


BYTECODE FOR CONTROL FLOW:
--------------------------

import dis

def conditional(x):
    if x > 0:
        return "Positive"
    else:
        return "Negative"

dis.dis(conditional)

# Output shows:
# COMPARE_OP: Compare two values
# POP_JUMP_IF_FALSE: Jump if condition is false
# LOAD_CONST: Load return value
# RETURN_VALUE: Return


BYTECODE FOR LOOPS:
-------------------

import dis

def loop():
    total = 0
    for i in range(5):
        total += i
    return total

dis.dis(loop)

# Output shows:

# LOAD_FAST: Load variable

# JUMP_ABSOLUTE: Jump back to loop start

3. PYTHON VIRTUAL MACHINE (PVM)
================================================================================

The Python Virtual Machine is a stack-based interpreter that executes bytecode
instructions.

STACK-BASED EXECUTION:
---------------------

# PVM uses a stack to execute operations
# Example: 5 + 3

import dis

def simple_add():
    return 5 + 3

dis.dis(simple_add)

# Execution:
# 1. LOAD_CONST 5 -> Push 5 onto stack: [5]
# 2. LOAD_CONST 3 -> Push 3 onto stack: [5, 3]
# 3. BINARY_ADD -> Pop 3 and 5, push 8: [8]
# 4. RETURN_VALUE -> Return 8


FRAME OBJECTS:
--------------

import sys

    print(f"Global variables count: {len(frame.f_globals)}")

show_frame_info()

code = "x = 5 + 3"
tree = ast.parse(code)

# Traverse AST

    print(type(node).__name__)

# Output:
# Module
# Assign
# Name
# BinOp
# Constant
# Add

COMPILATION STEPS:
------------------

# 1. Tokenization: Source -> Tokens
# 2. Parsing: Tokens -> AST
# 3. Compilation: AST -> Bytecode
# 4. Execution: Bytecode -> Results

import tokenize
import io

code = "x = 5 + 3"
tokens = tokenize.generate_tokens(io.StringIO(code).readline)

for token in tokens:
    print(token)


COMPILE FUNCTION:
-----------------

# Compile source code to code object
code_str = """
def greet(name):
    return f"Hello, {name}!"
"""

code_obj = compile(code_str, filename="<string>", mode="exec")

# Execute compiled code
namespace = {}
exec(code_obj, namespace)

# Use the function
greet = namespace["greet"]
print(greet("Alice"))  # Output: Hello, Alice!

-------------

CIRCULAR IMPORTS:
-----------------

# module_a.py
# from module_b import function_b
# def function_a():
#     return function_b()

# module_b.py
# from module_a import function_a
# def function_b():
#     return function_a()

# This causes circular import error
# Solution: Import inside functions or use late binding

6. OPTIMIZATION TECHNIQUES
================================================================================

FUNCTION CALL OPTIMIZATION:
---------------------------

import timeit

# Local variable access is faster than global
def with_local():
    x = 10
    for i in range(1000):
        x += i
    return x

def with_global():
    global x
    x = 10
    for i in range(1000):
        x += i
    return x

# Local is faster
print(timeit.timeit(with_local, number=10000))
print(timeit.timeit(with_global, number=10000))


LOOP OPTIMIZATION:
------------------

import timeit

# Avoid attribute lookup in loops
class Counter:
    def __init__(self):
        self.count = 0
    
    def increment(self):
        self.count += 1

counter = Counter()

# Slow: attribute lookup in each iteration
def slow_loop():
    for i in range(1000):
        counter.increment()

# Fast: cache method reference
def fast_loop():
    increment = counter.increment
    for i in range(1000):
        increment()

print(timeit.timeit(slow_loop, number=1000))
print(timeit.timeit(fast_loop, number=1000))


BUILT-IN FUNCTIONS:
-------------------

import timeit

# Built-in functions are optimized in C
# Use them instead of Python equivalents

# Slow: Python loop
def slow_sum(numbers):
    total = 0
    for num in numbers:
        total += num
    return total

# Fast: Built-in function
def fast_sum(numbers):
    return sum(numbers)

numbers = list(range(1000))
print(timeit.timeit(lambda: slow_sum(numbers), number=1000))
print(timeit.timeit(lambda: fast_sum(numbers), number=1000))


LIST COMPREHENSION:
-------------------

import timeit

# List comprehension is faster than append loop
numbers = list(range(1000))

def with_append():
    result = []
    for num in numbers:
        result.append(num * 2)
    return result

def with_comprehension():
    return [num * 2 for num in numbers]

print(timeit.timeit(with_append, number=1000))
print(timeit.timeit(with_comprehension, number=1000))

EXAMPLE 1: ANALYZING FUNCTION PERFORMANCE
------------------------------------------

import dis
import timeit

def fibonacci_recursive(n):
    if n < 2:
        return n
    return fibonacci_recursive(n-1) + fibonacci_recursive(n-2)

def fibonacci_iterative(n):
    a, b = 0, 1
    for _ in range(n):
        a, b = b, a + b
    return a

# Compare bytecode
print("Recursive:")
dis.dis(fibonacci_recursive)

print("\nIterative:")
dis.dis(fibonacci_iterative)

# Compare performance
print(f"\nRecursive: {timeit.timeit(lambda: fibonacci_recursive(20), number=1):.4f}s")
print(f"Iterative: {timeit.timeit(lambda: fibonacci_iterative(20), number=1000):.4f}s")


EXAMPLE 2: UNDERSTANDING EXECUTION FLOW
----------------------------------------

import sys

class ExecutionTracer:
    def __init__(self):
        self.depth = 0
    
    def trace_calls(self, frame, event, arg):
        if event == 'call':
            code = frame.f_code
            print("  " * self.depth + f"-> {code.co_name}")
            self.depth += 1
        elif event == 'return':
            self.depth -= 1
        return self.trace_calls

def function_a():
    function_b()

def function_b():
    function_c()

def function_c():
    return 42

tracer = ExecutionTracer()
sys.settrace(tracer.trace_calls)
function_a()
sys.settrace(None)

2. Bytecode is platform-independent and cached in __pycache__
3. Python Virtual Machine is stack-based and executes bytecode
4. Use dis module to view bytecode and understand execution

6. Built-in functions are optimized and faster than Python equivalents
