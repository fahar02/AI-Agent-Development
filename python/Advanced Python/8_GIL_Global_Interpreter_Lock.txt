1. INTRODUCTION TO GIL
================================================================================

The Global Interpreter Lock (GIL) is a mutex that protects access to Python
objects in CPython. It prevents multiple threads from executing Python code
simultaneously within a single process.

Key Points:
- Only one thread can execute Python bytecode at a time
- Applies only to CPython (not Jython, IronPython, PyPy)
- Exists for memory management simplicity
- Can be released for I/O operations
- Affects CPU-bound but not I/O-bound tasks


================================================================================
2. WHAT IS THE GIL?
================================================================================

WHY DOES GIL EXIST?
-------------------

CPython uses reference counting for memory management. Without the GIL:
- Every object would need its own lock
- Significant performance overhead
- Complex deadlock prevention
- Slower single-threaded performance

The GIL is a trade-off: simpler implementation vs. limited parallelism.


GIL MECHANICS:
--------------

# The GIL is released periodically
# By default, every 5ms (sys.getswitchinterval())

import sys

# Check GIL switch interval
print(f"GIL switch interval: {sys.getswitchinterval()} seconds")

# Change GIL switch interval
sys.setswitchinterval(0.01)  # 10ms

# GIL is released during:
# - I/O operations (file, network, database)
# - C extension calls
# - Some built-in operations


CHECKING GIL STATUS:
--------------------

import sys

# Check if GIL is enabled
# In CPython, GIL is always enabled
# In other implementations, it might not be

# You can't directly check GIL status, but you can observe its effects
import threading
import time

def cpu_bound_task():
    total = 0
    for i in range(50000000):
        total += i
    return total

# Single-threaded
start = time.time()
cpu_bound_task()
single_time = time.time() - start
print(f"Single-threaded: {single_time:.2f}s")

# Multi-threaded (will be slower due to GIL)
start = time.time()
threads = []
for _ in range(2):
    t = threading.Thread(target=cpu_bound_task)
    threads.append(t)
    t.start()
for t in threads:
    t.join()
multi_time = time.time() - start
print(f"Multi-threaded: {multi_time:.2f}s")
print(f"Speedup: {single_time / multi_time:.2f}x (should be <1 due to GIL)")

3. IMPACT ON THREADING
================================================================================

THREADING WITH CPU-BOUND TASKS:
-------------------------------

import threading
import time

def cpu_bound_work(duration=5):
    end_time = time.time() + duration
    count = 0
    while time.time() < end_time:
        count += 1
    return count

# Single thread
start = time.time()
result = cpu_bound_work(5)
single_time = time.time() - start
print(f"Single thread: {single_time:.2f}s, count: {result}")

# Two threads (GIL prevents true parallelism)
start = time.time()
threads = []
results = []
for _ in range(2):
    t = threading.Thread(target=lambda: results.append(cpu_bound_work(5)))
    threads.append(t)
    t.start()
for t in threads:
    t.join()
multi_time = time.time() - start
print(f"Two threads: {multi_time:.2f}s")


THREADING WITH I/O-BOUND TASKS:
-------------------------------

import threading
import time

def io_bound_work(url):
    # I/O releases the GIL
    time.sleep(1)
    return f"Data from {url}"

urls = ["url1", "url2", "url3"]

# Sequential (slow)
start = time.time()
for url in urls:
    io_bound_work(url)
sequential_time = time.time() - start
print(f"Sequential: {sequential_time:.2f}s")

# Threaded (fast - GIL released during I/O)
start = time.time()
threads = []
for url in urls:
    t = threading.Thread(target=io_bound_work, args=(url,))
    threads.append(t)
    t.start()
for t in threads:
    t.join()
threaded_time = time.time() - start
print(f"Threaded: {threaded_time:.2f}s")
print(f"Speedup: {sequential_time / threaded_time:.2f}x")


THREAD SYNCHRONIZATION:
-----------------------

import threading

counter = 0
lock = threading.Lock()

def increment_without_lock():
    global counter
    for _ in range(1000000):
        counter += 1

def increment_with_lock():
    global counter
    for _ in range(1000000):
        with lock:
            counter += 1

# Without lock (race condition)
counter = 0
threads = [threading.Thread(target=increment_without_lock) for _ in range(2)]
for t in threads:
    t.start()
for t in threads:
    t.join()
print(f"Without lock: {counter} (expected 2000000)")

# With lock (correct)
counter = 0
threads = [threading.Thread(target=increment_with_lock) for _ in range(2)]
for t in threads:
    t.start()
for t in threads:
    t.join()
print(f"With lock: {counter} (expected 2000000)")

4. IMPACT ON MULTIPROCESSING
================================================================================

MULTIPROCESSING VS THREADING:
-----------------------------

import multiprocessing
import threading
import time

def cpu_bound_task(n):
    total = 0
    for i in range(n):
        total += i
    return total

n = 50000000

# Single process
start = time.time()
cpu_bound_task(n)
single_time = time.time() - start
print(f"Single process: {single_time:.2f}s")

# Threading (limited by GIL)
start = time.time()
threads = []
for _ in range(2):
    t = threading.Thread(target=cpu_bound_task, args=(n,))
    threads.append(t)
    t.start()
for t in threads:
    t.join()
thread_time = time.time() - start
print(f"Threading: {thread_time:.2f}s")

# Multiprocessing (true parallelism)
start = time.time()
processes = []
for _ in range(2):
    p = multiprocessing.Process(target=cpu_bound_task, args=(n,))
    processes.append(p)
    p.start()
for p in processes:
    p.join()
process_time = time.time() - start
print(f"Multiprocessing: {process_time:.2f}s")


PROCESS POOL:
-------------

import multiprocessing

def square(x):
    return x ** 2

# Using process pool
with multiprocessing.Pool(processes=4) as pool:
    results = pool.map(square, range(10))
    print(results)


SHARED DATA IN MULTIPROCESSING:
-------------------------------

import multiprocessing

def worker(shared_list, index, value):
    shared_list[index] = value

if __name__ == "__main__":
    # Shared list
    shared_list = multiprocessing.Manager().list([0] * 5)
    
    processes = []
    for i in range(5):
        p = multiprocessing.Process(target=worker, args=(shared_list, i, i*2))
        processes.append(p)
        p.start()
    
    for p in processes:
        p.join()
    
    print(list(shared_list))

5. GIL WORKAROUNDS
================================================================================

WORKAROUND 1: USE MULTIPROCESSING FOR CPU-BOUND TASKS
-----------------------------------------------------

import multiprocessing
import time

def cpu_intensive(n):
    total = 0
    for i in range(n):
        total += i
    return total

if __name__ == "__main__":
    # Use multiprocessing for CPU-bound work
    with multiprocessing.Pool(processes=4) as pool:
        results = pool.map(cpu_intensive, [50000000] * 4)
    print(f"Results: {results}")


WORKAROUND 2: USE THREADING FOR I/O-BOUND TASKS
-----------------------------------------------

import threading
import time

def io_task(task_id):
    # Simulate I/O operation
    time.sleep(1)
    print(f"Task {task_id} completed")

# Threading works well for I/O
threads = []
for i in range(5):
    t = threading.Thread(target=io_task, args=(i,))
    threads.append(t)
    t.start()

for t in threads:
    t.join()


WORKAROUND 3: USE ASYNCIO FOR I/O-BOUND TASKS
----------------------------------------------

import asyncio

async def io_task(task_id):
    # Simulate I/O operation
    await asyncio.sleep(1)
    print(f"Task {task_id} completed")

async def main():
    # Run multiple tasks concurrently
    await asyncio.gather(*[io_task(i) for i in range(5)])

asyncio.run(main())


WORKAROUND 4: USE C EXTENSIONS
-------------------------------

# NumPy operations release the GIL
import numpy as np
import threading
import time

def numpy_work():
    a = np.random.rand(1000, 1000)
    b = np.random.rand(1000, 1000)
    c = np.dot(a, b)
    return c

# NumPy operations are fast in threads because GIL is released
start = time.time()
threads = []
for _ in range(2):
    t = threading.Thread(target=numpy_work)
    threads.append(t)
    t.start()
for t in threads:
    t.join()
print(f"NumPy threading: {time.time() - start:.2f}s")

COMPARISON TABLE:

-----------------------------------------------

import asyncio

async def fetch_data(url):
    await asyncio.sleep(0.1)
    return f"Data from {url}"

async def main():
    tasks = [fetch_data(f"url{i}") for i in range(100)]
    results = await asyncio.gather(*tasks)

asyncio.run(main())


PRACTICE 4: AVOID SHARED MUTABLE STATE
--------------------------------------

# Bad: Shared mutable state
counter = 0
lock = threading.Lock()

def increment():
    global counter
    with lock:
        counter += 1

# Good: Use thread-safe data structures
import queue
q = queue.Queue()

def producer():
    q.put("item")

def consumer():
    item = q.get()


PRACTICE 5: USE THREAD POOLS FOR MANY THREADS
----------------------------------------------

from concurrent.futures import ThreadPoolExecutor

def task(n):
    return n ** 2

# Use thread pool instead of creating many threads
with ThreadPoolExecutor(max_workers=4) as executor:
    results = executor.map(task, range(10))

8. REAL-WORLD EXAMPLES
================================================================================

EXAMPLE 1: WEB SCRAPER WITH THREADING
--------------------------------------

import threading
import time
from queue import Queue

def scrape_url(url, results_queue):
    # Simulate scraping
    time.sleep(1)
    results_queue.put(f"Scraped {url}")

urls = [f"url{i}" for i in range(10)]
results_queue = Queue()

# Use thread pool
threads = []
for url in urls:
    t = threading.Thread(target=scrape_url, args=(url, results_queue))
    threads.append(t)
    t.start()

for t in threads:
    t.join()

while not results_queue.empty():
    print(results_queue.get())


EXAMPLE 2: CPU-INTENSIVE TASK WITH MULTIPROCESSING
---------------------------------------------------

import multiprocessing

def calculate_prime(n):
    count = 0
    for i in range(2, n):
        is_prime = True
        for j in range(2, int(i**0.5) + 1):
            if i % j == 0:
                is_prime = False
                break
        if is_prime:
            count += 1
    return count

if __name__ == "__main__":
    with multiprocessing.Pool(processes=4) as pool:
        results = pool.map(calculate_prime, [10000, 20000, 30000, 40000])
    print(f"Prime counts: {results}")


EXAMPLE 3: ASYNC WEB SERVER
----------------------------

import asyncio

async def handle_request(request_id):
    await asyncio.sleep(0.1)
    return f"Response {request_id}"

async def main():
    # Handle 1000 requests concurrently
    tasks = [handle_request(i) for i in range(1000)]
    results = await asyncio.gather(*tasks)
    print(f"Handled {len(results)} requests")

asyncio.run(main())


================================================================================
KEY TAKEAWAYS
================================================================================

1. GIL prevents true parallelism in CPython for CPU-bound tasks
2. GIL is released during I/O operations, making threading viable for I/O
3. Use threading for I/O-bound tasks
4. Use multiprocessing for CPU-bound tasks
5. Use asyncio for many concurrent I/O operations
6. Avoid shared mutable state between threads
7. Use thread-safe data structures like Queue
8. C extensions can release the GIL
9. NumPy operations release the GIL
10. Choose the right concurrency model for your task type
