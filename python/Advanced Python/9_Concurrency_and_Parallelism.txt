1. INTRODUCTION TO CONCURRENCY AND PARALLELISM
================================================================================

CONCURRENCY VS PARALLELISM:
---------------------------

Concurrency:
- Multiple tasks appear to run simultaneously
- Actually interleave execution on single core
- Useful for I/O-bound tasks
- Lower overhead

Parallelism:
- Multiple tasks run simultaneously on multiple cores
- True parallel execution
- Useful for CPU-bound tasks
- Higher overhead

WHEN TO USE EACH:
-----------------

Concurrency (Threading/Asyncio):
- I/O-bound tasks (network, file, database)
- Many lightweight tasks
- Responsive user interfaces
- Server handling multiple clients

Parallelism (Multiprocessing):
- CPU-bound tasks (calculations, processing)
- Heavy computational work
- Data processing pipelines
- Scientific computing

BASIC THREADING:
----------------

import threading
import time

def worker(name, duration):
    print(f"Worker {name} started")
    time.sleep(duration)
    print(f"Worker {name} finished")

# Create threads
threads = []
for i in range(3):
    t = threading.Thread(target=worker, args=(f"Thread-{i}", 1))
    threads.append(t)
    t.start()

# Wait for all threads
for t in threads:
    t.join()

print("All threads completed")


THREAD CLASS:
-------------

import threading
import time

class WorkerThread(threading.Thread):
    def __init__(self, name, duration):
        super().__init__()
        self.name = name
        self.duration = duration
    
    def run(self):
        print(f"{self.name} started")
        time.sleep(self.duration)
        print(f"{self.name} finished")

# Create and start threads
threads = []
for i in range(3):
    t = WorkerThread(f"Worker-{i}", 1)
    threads.append(t)
    t.start()

# Wait for completion
for t in threads:
    t.join()


THREAD SYNCHRONIZATION:
-----------------------

import threading

counter = 0
lock = threading.Lock()

def increment():
    global counter
    for _ in range(100000):
        with lock:
            counter += 1

threads = [threading.Thread(target=increment) for _ in range(5)]
for t in threads:
    t.start()
for t in threads:
    t.join()

print(f"Counter: {counter}")  # Output: 500000


THREAD-SAFE QUEUE:
------------------

import threading
import queue
import time

def producer(q):
    for i in range(5):
        q.put(f"Item {i}")
        time.sleep(0.1)

def consumer(q):
    while True:
        item = q.get()
        if item is None:
            break
        print(f"Consumed: {item}")
        q.task_done()

q = queue.Queue()

# Create producer and consumer threads
prod_thread = threading.Thread(target=producer, args=(q,))
cons_thread = threading.Thread(target=consumer, args=(q,))

prod_thread.start()
cons_thread.start()

prod_thread.join()
q.put(None)  # Signal consumer to stop
cons_thread.join()


THREAD POOL:
------------

from concurrent.futures import ThreadPoolExecutor

    time.sleep(1)
    return n ** 2

# Using thread pool
with ThreadPoolExecutor(max_workers=3) as executor:
    results = executor.map(task, range(5))
    for result in results:
        print(result)

import multiprocessing

def producer(queue):
    for i in range(5):
        queue.put(f"Item {i}")

def consumer(queue):
    while True:
        item = queue.get()

        print(f"Received: {item}")

ASYNC CONTEXT MANAGER:
----------------------

import asyncio

class AsyncResource:
    async def __aenter__(self):
        print("Acquiring resource")
        await asyncio.sleep(0.1)
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        print("Releasing resource")
        await asyncio.sleep(0.1)
    
    async def do_work(self):
        print("Doing work")
        await asyncio.sleep(0.5)

async def main():
    async with AsyncResource() as resource:
        await resource.do_work()

asyncio.run(main())


ASYNC ITERATOR:
---------------

import asyncio

class AsyncCounter:
    def __init__(self, max):
        self.max = max
        self.current = 0
    
    def __aiter__(self):
        return self
    
    async def __anext__(self):
        if self.current < self.max:
            self.current += 1
            await asyncio.sleep(0.1)
            return self.current
        raise StopAsyncIteration

async def main():
    async for num in AsyncCounter(5):
        print(num)

asyncio.run(main())

5. ASYNCIO

EVENT LOOP:
-----------

import asyncio

async def task1():
    print("Task 1 started")
    await asyncio.sleep(1)
    print("Task 1 finished")

async def task2():
    print("Task 2 started")
    await asyncio.sleep(0.5)
    print("Task 2 finished")

async def main():
    # Create tasks
    t1 = asyncio.create_task(task1())
    t2 = asyncio.create_task(task2())
    
    # Wait for all tasks
    await asyncio.gather(t1, t2)

asyncio.run(main())


ASYNCIO QUEUE:
--------------

import asyncio

async def producer(queue):
    for i in range(5):
        await queue.put(f"Item {i}")
        await asyncio.sleep(0.1)

async def consumer(queue):
    while True:
        item = await queue.get()
        print(f"Consumed: {item}")
        queue.task_done()

async def main():
    queue = asyncio.Queue()
    
    prod_task = asyncio.create_task(producer(queue))
    cons_task = asyncio.create_task(consumer(queue))
    
    await prod_task
    await queue.join()
    cons_task.cancel()

asyncio.run(main())


ASYNCIO LOCK:
-------------

import asyncio

counter = 0
lock = asyncio.Lock()

async def increment():
    global counter
    for _ in range(100):
        async with lock:
            counter += 1

async def main():
    await asyncio.gather(increment(), increment(), increment())
    print(f"Counter: {counter}")

asyncio.run(main())


ASYNCIO SEMAPHORE:
------------------

import asyncio

async def task(semaphore, task_id):
    async with semaphore:
        print(f"Task {task_id} started")
        await asyncio.sleep(1)
        print(f"Task {task_id} finished")

async def main():
    semaphore = asyncio.Semaphore(2)  # Max 2 concurrent tasks
    
    tasks = [task(semaphore, i) for i in range(5)]
    await asyncio.gather(*tasks)

asyncio.run(main())

6. COMPARISON AND USE CASES
================================================================================

THREADING VS MULTIPROCESSING VS ASYNCIO:
----------------------------------------

Use Threading when:
- I/O-bound tasks (network, file, database)
- Need shared memory
- Want simple implementation
- Have moderate number of concurrent tasks

Use Multiprocessing when:
- CPU-bound tasks (calculations, processing)
- Need true parallelism
- Can afford process overhead
- Tasks are independent

Use Asyncio when:
- Many I/O-bound tasks (1000+)
- Need minimal overhead
- Can use async/await syntax
- Single-threaded is acceptable


PERFORMANCE COMPARISON:
-----------------------

import threading
import multiprocessing
import asyncio
import time

def cpu_task(n):
    total = 0
    for i in range(n):
        total += i
    return total

async def io_task():
    await asyncio.sleep(0.1)
    return "done"

# CPU-bound: Multiprocessing wins
# I/O-bound: Asyncio wins
# Shared state: Threading wins

TASK CANCELLATION:
------------------

import asyncio

async def long_task():
    try:
        for i in range(10):
            print(f"Step {i}")
            await asyncio.sleep(1)
    except asyncio.CancelledError:
        print("Task cancelled")
        raise

async def main():
    task = asyncio.create_task(long_task())
    
    await asyncio.sleep(3)
    task.cancel()
    
    try:
        await task
    except asyncio.CancelledError:
        print("Task was cancelled")

asyncio.run(main())

EXAMPLE 5: RATE LIMITER WITH ASYNCIO
-------------------------------------

import asyncio
from datetime import datetime, timedelta

class RateLimiter:
    def __init__(self, max_calls, time_window):
        self.max_calls = max_calls
        self.time_window = time_window
        self.calls = []
    
    async def acquire(self):
        now = datetime.now()
        # Remove old calls outside time window
        self.calls = [call for call in self.calls 
                     if now - call < timedelta(seconds=self.time_window)]
        
        if len(self.calls) >= self.max_calls:
            sleep_time = (self.calls[0] + timedelta(seconds=self.time_window) - now).total_seconds()
            await asyncio.sleep(sleep_time)
            return await self.acquire()
        
        self.calls.append(now)

async def api_call(limiter):
    await limiter.acquire()
    print(f"API call at {datetime.now()}")

async def main():
    limiter = RateLimiter(max_calls=3, time_window=1)
    
    tasks = [api_call(limiter) for _ in range(10)]
    await asyncio.gather(*tasks)

# asyncio.run(main())


================================================================================
KEY TAKEAWAYS
================================================================================

1. Use threading for I/O-bound tasks with shared state
2. Use multiprocessing for CPU-bound tasks
3. Use asyncio for many concurrent I/O operations
4. Threading has lower overhead than multiprocessing
5. Asyncio is more efficient than threading for many tasks
6. Always use thread/process pools for better resource management
7. Use thread-safe data structures (Queue) for inter-thread communication
8. Use asyncio.gather() to run multiple coroutines concurrently
9. Handle exceptions properly in concurrent code
10. Choose the right concurrency model based on task type and requirements
